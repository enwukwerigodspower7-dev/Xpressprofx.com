<script>
  /* ── MARKET DATA (simulated) ── */
  const marketData = [
    { sym: 'AAPL', name: 'Apple Inc.', price: 187.42, change: +1.23 },
    { sym: 'TSLA', name: 'Tesla Inc.', price: 242.18, change: -3.71 },
    { sym: 'NVDA', name: 'NVIDIA Corp.', price: 875.60, change: +12.40 },
    { sym: 'BTC', name: 'Bitcoin', price: 68420.00, change: +2.14 },
    { sym: 'ETH', name: 'Ethereum', price: 3580.00, change: -1.08 },
    { sym: 'EUR/USD', name: 'Euro / US Dollar', price: 1.0854, change: +0.12 },
    { sym: 'GBP/USD', name: 'British Pound', price: 1.2698, change: -0.23 },
    { sym: 'GOLD', name: 'Gold Spot (XAU)', price: 2318.50, change: +0.45 },
  ];

  /* ── TICKER TAPE ── */
  function buildTicker() {
    const track = document.getElementById('tickerTrack');
    const doubled = [...marketData, ...marketData];
    doubled.forEach(d => {
      const isUp = d.change >= 0;
      const el = document.createElement('div');
      el.className = 'ticker-item';
      el.innerHTML = `
        <span class="sym">${d.sym}</span>
        <span class="price">${formatPrice(d.sym, d.price)}</span>
        <span class="${isUp ? 'up' : 'down'}">${isUp ? '▲' : '▼'} ${Math.abs(d.change)}%</span>
      `;
      track.appendChild(el);
    });
  }

  function formatPrice(sym, price) {
    if (sym === 'BTC') return '$' + price.toLocaleString();
    if (sym === 'ETH') return '$' + price.toLocaleString();
    if (sym.includes('/')) return price.toFixed(4);
    return '$' + price.toFixed(2);
  }

  /* ── MARKET CARDS ── */
  function buildMarketCards() {
    const grid = document.getElementById('marketGrid');
    marketData.forEach(d => {
      const isUp = d.change >= 0;
      const card = document.createElement('div');
      card.className = 'market-card';
      card.innerHTML = `
        <div class="market-card-header">
          <span class="market-symbol">${d.sym}</span>
          <span class="market-badge ${isUp ? 'badge-up' : 'badge-down'}">
            ${isUp ? '▲' : '▼'} ${Math.abs(d.change)}%
          </span>
        </div>
        <div class="market-price">${formatPrice(d.sym, d.price)}</div>
        <div class="market-name">${d.name}</div>
        <svg class="mini-chart" viewBox="0 0 200 48" preserveAspectRatio="none">
          <defs>
            <linearGradient id="g${d.sym.replace('/','')}" x1="0" y1="0" x2="0" y2="1">
              <stop offset="0%" stop-color="${isUp ? '#00c896' : '#ef4444'}" stop-opacity="0.4"/>
              <stop offset="100%" stop-color="${isUp ? '#00c896' : '#ef4444'}" stop-opacity="0"/>
            </linearGradient>
          </defs>
          <path d="${generateSparkline(isUp)}" fill="url(#g${d.sym.replace('/','')})"/>
          <path d="${generateSparklineLine(isUp)}" fill="none" stroke="${isUp ? '#00c896' : '#ef4444'}" stroke-width="2"/>
        </svg>
      `;
      card.onclick = () => openModal('register');
      grid.appendChild(card);
    });
  }

  function generateSparkline(up) {
    const points = generatePoints(up);
    const path = points.map((p, i) => `${i === 0 ? 'M' : 'L'}${p.x},${p.y}`).join(' ');
    const last = points[points.length - 1];
    return `${path} L${last.x},48 L0,48 Z`;
  }

  function generateSparklineLine(up) {
    const points = generatePoints(up);
    return points.map((p, i) => `${i === 0 ? 'M' : 'L'}${p.x},${p.y}`).join(' ');
  }

  function generatePoints(up) {
    const pts = [];
    let y = up ? 35 : 15;
    for (let i = 0; i <= 10; i++) {
      y += (Math.random() - (up ? 0.35 : 0.65)) * 12;
      y = Math.max(5, Math.min(43, y));
      pts.push({ x: i * 20, y });
    }
    return pts;
  }

  /* ── LIVE PRICE UPDATES ── */
  function updatePrices() {
    marketData.forEach(d => {
      const delta = (Math.random() - 0.5) * 0.002;
      d.price = d.price * (1 + delta);
      d.change = parseFloat((d.change + (Math.random() - 0.5) * 0.05).toFixed(2));
    });
    // Rebuild cards
    const grid = document.getElementById('marketGrid');
    grid.innerHTML = '';
    buildMarketCards();
  }

  setInterval(updatePrices, 4000);

  /* ── MODAL ── */
  function openModal(tab = 'login') {
    document.getElementById('authModal').classList.add('open');
    switchTab(tab);
    document.body.style.overflow = 'hidden';
  }

  function closeModal() {
    document.getElementById('authModal').classList.remove('open');
    document.body.style.overflow = '';
  }

  function closeModalOutside(e) {
    if (e.target === document.getElementById('authModal')) closeModal();
  }

  function switchTab(tab) {
    const isLogin = tab === 'login';
    document.getElementById('loginTab').classList.toggle('active', isLogin);
    document.getElementById('registerTab').classList.toggle('active', !isLogin);
    document.getElementById('loginPanel').classList.toggle('active', isLogin);
    document.getElementById('registerPanel').classList.toggle('active', !isLogin);
    document.getElementById('modalTitle').textContent = isLogin ? 'Welcome Back' : 'Create Your Account';
    document.getElementById('modalSub').textContent = isLogin ? 'Log in to your ApexTrade account.' : 'Join 500,000+ investors worldwide.';
  }

  function handleAuth(type) {
    alert(`✅ ${type === 'login' ? 'Login' : 'Registration'} flow would proceed here.\n\nIn a real implementation, this would connect to your auth backend (e.g., Firebase, Auth0, or custom API).`);
    closeModal();
  }

  function handleContactSubmit(e) {
    e.preventDefault();
    alert('✅ Message received! Our team will get back to you within 24 hours.');
  }

  /* ── MOBILE NAV ── */
  function toggleMenu() {
    document.getElementById('navLinks').classList.toggle('open');
    document.getElementById('navCta').classList.toggle('open');
  }

  /* ── SCROLL EVENTS ── */
  window.addEventListener('scroll', () => {
    const scrollBtn = document.getElementById('scrollTop');
    if (window.scrollY > 400) scrollBtn.classList.add('visible');
    else scrollBtn.classList.remove('visible');
  });

  /* ── INTERSECTION OBSERVER (Fade-up) ── */
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
      }
    });
  }, { threshold: 0.1 });

  document.querySelectorAll('.fade-up').forEach(el => observer.observe(el));

  /* ── KEYBOARD CLOSE ── */
  document.addEventListener('keydown', e => {
    if (e.key === 'Escape') closeModal();
  });

  /* ── INIT ── */
  buildTicker();
  buildMarketCards();
